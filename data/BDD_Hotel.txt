CREATE TABLE CATEGORIE(
  NUCAT INT PRIMARY KEY,
  NOMCAT VARCHAR(100)
);


CREATE TABLE TYPE_HOTEL(
  NUTYPE INT PRIMARY KEY,
  NOMTYPE VARCHAR(100)
);


CREATE TABLE HOTEL(
  NUHOTEL       INT PRIMARY KEY,
  NUTYPEHOTEL   INT,
  NOMHOTEL      VARCHAR(100),
  ADRESSEHOTEL  VARCHAR(100),
  CPHOTEL       VARCHAR(10),
  VILLEHOTEL    VARCHAR(50),
  
  CONSTRAINT FK_NUTYPEHOTEL FOREIGN KEY (NUTYPEHOTEL) REFERENCES TYPE_HOTEL(NUTYPE)
  
);


CREATE TABLE CHAMBRE(
  NUCHAMBRE     INT,
  NUCATCHAMBRE  INT,
  NUHOTEL       INT,
  CONSTRAINT FK_NUHOTEL2     FOREIGN KEY (NUHOTEL)     REFERENCES HOTEL(NUHOTEL),
  CONSTRAINT FK_NUCATHOTEL2  FOREIGN KEY (NUCATCHAMBRE)  REFERENCES CATEGORIE(NUCAT)
  
);
